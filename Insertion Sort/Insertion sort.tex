        \begin{algorithm}
            \caption{INSERTION SORT}
            \begin{algorithmic}[1]
                \Require $A$: Array with $n$ elements
                \Ensure Ascending Array
                \For{$i \gets 2 $ \bf{to} $n$}% \to $n$}
                    \State $key \gets A[i]$
                    \State $j \gets i - 1$
                    \While{$j>0$ and $A[j]>key$}
                        \State $A[j + 1] \gets A[j]$
                        \State $j \gets j - 1$
                    \EndWhile
                    \State $A[j + 1] \gets key$
                \EndFor
            \end{algorithmic}
        \end{algorithm}
    \end{CJK*}
